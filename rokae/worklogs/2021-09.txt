import matplotlib.pyplot as plt
import numpy as np

x = np.arange(-1, 1, 0.02)
y = 2 * np.sin(x * 2.3) + np.random.rand(len(x))

parameter = np.polyfit(x, y, 3)
y2 = parameter[0] * x ** 3 + parameter[1] * \
    x ** 2 + parameter[2] * x + parameter[3]
plt.scatter(x, y)
plt.plot(x, y2, color='g')
plt.show()
